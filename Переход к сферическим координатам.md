$$\det\phi'(\rho,\varphi,\theta)=
\begin{vmatrix}
\cos\varphi\sin\theta & -\rho\sin\varphi\sin\theta & \rho\cos\varphi\cos\theta \\
\sin\varphi\sin\theta & \rho\cos\varphi\sin\theta & \rho\sin\varphi\cos\theta \\
\cos\theta & 0 & -\rho\sin\theta
\end{vmatrix}=
-\rho^2\sin\theta
$$
